	0 argument : -> switch() -> case 0 -> f1() -> if() -> switch() -> case 1

	2 arguments : -> switch() -> case 0 -> f1() -> f2() -> switch() -> case 2
	+2 arguments : -> switch() -> case 0 -> f1() -> f2() -> switch() -> case 2
